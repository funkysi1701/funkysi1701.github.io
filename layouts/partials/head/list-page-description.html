{{- /* Home, section, and taxonomy list pages: site tagline plus optional pagination suffix. */ -}} {{- $page := . -}} {{- $title := $page.Title -}} {{- if and $page.IsHome (eq $title "") -}} {{- $title = $page.Site.Title -}} {{- end -}} {{- $pageDesc := printf "%s - %s" $title ($page.Site.Params.description | default "") | plainify | htmlUnescape -}} {{- $supportsPagination := or $page.IsHome (eq $page.Kind "taxonomy") (eq $page.Kind "term") (eq $page.Kind "section") -}} {{- if $supportsPagination -}} {{- with $page.Paginator -}} {{- if gt .PageNumber 1 -}} {{- $pageDesc = printf "%s %s" $pageDesc (i18n "pagination_title" .) -}} {{- end -}} {{- end -}} {{- end -}} {{- return $pageDesc -}}
